struct nf_conntrack_tuple *target,enum nf_nat_manip_type maniptype)

{    

struct iphdr *iph;

const struct nf_nat_protocol *p;

if (!skb_make_writable(skb, iphdroff + sizeof(*iph)))

  return false;

iph = (void *)skb->data + iphdroff;

 p = __nf_nat_proto_find(proto); /*查找出具体的协议操作集*/

      /*完成传输层头的端口转换，校验和的修改*/

 if (!p->manip_pkt(skb, iphdroff, target, maniptype))

return false;

 

iph = (void *)skb->data + iphdroff;

if (maniptype == IP_NAT_MANIP_SRC) {

csum_replace4(&iph->check, iph->saddr, target->src.u3.ip);/*进行网络层头校验和修改*/

iph->saddr = target->src.u3.ip; /*进行网络层头中ip地址的转换*/

} else {

csum_replace4(&iph->check, iph->daddr, target->dst.u3.ip);

iph->daddr = target->dst.u3.ip;

}

return true;

}
