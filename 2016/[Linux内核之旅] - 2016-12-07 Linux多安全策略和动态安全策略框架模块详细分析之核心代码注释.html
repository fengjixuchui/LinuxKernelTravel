对于访问向量缓存，其中最核心的函数为avc_search_node()和avs_has_perm_noaudit()，下面针对其进行详细解释

1. avc_search_node()

static inline struct avc_node *avc_search_node(u32 ssid, u32 tsid, u16 tclass)

{

        struct avc_node *node, *ret = NULL;

        int hvalue;

        struct hlist_head *head;

        struct hlist_node *next;

  /*获取参数指定的访问向量在哈希表中的哈希值*/

        hvalue = avc_hash(ssid, tsid, tclass);

        head = &avc_cache.slots[hvalue];

  /*

   * 调用hlist_for_each_entry_rcu()宏来遍历具有相同哈希值的访问向量组成的链表，如果某个访问向量中指定的源SID、目标SID、目标类别均与待查找的访问向量相同，则该访问向量即为要查找的访问向量，然后返回该节点。*/

        hlist_for_each_entry_rcu(node, next, head, list) {

                if (ssid == node->ae.ssid &&

                  tclass == node->ae.tclass &&

                  tsid == node->ae.tsid) {

                       ret = node;

                       break;

                }

        }

        return ret;

 }

2 avc_has_perm_noaudit()

 

inline int avc_has_perm_noaudit(u32 ssid, u32 tsid,u16 tclass, u32 requested,unsigned flags,

                           struct av_decision *avd)

{

       struct avc_node *node;

       int rc = 0;

       u32 denied;

       BUG_ON(!requested);

       rcu_read_lock();

    /*

* 在访问向量缓存中查找指定的SID对对应的访问向量结点，并将查找结果存放到node中。*/

       node = avc_lookup(ssid, tsid, tclass);

   /*

 * 判断node是否为空，如果为空，则调用avc_compute_av()函数来在安全服务器查找对应的访问向量，并将找到的访问向量插入到访问向量缓存中；否则将找到的访问向量结点中的访问向量决策拷贝到参数avd中。*/

       if (unlikely(!node)) {

              node = avc_compute_av(ssid, tsid, tclass, avd);

       } else {

              memcpy(avd, &node->ae.avd, sizeof(*avd));

          avd = &node->ae.avd;

    }

   /*

* 判断是否请求了访问向量未授予的权限，若是，则调用avc_denied()函数将相应* 的权限授予SID对。*/

       denied = requested & ~(avd->allowed);

       if (unlikely(denied))

               rc = avc_denied(ssid, tsid, tclass, requested, flags, avd);

       rcu_read_unlock();

       return rc;

}
