//查找一条规则

static struct audit_entry *audit_find_rule(struct audit_entry *entry, struct list_head **p)

{

struct audit_entry *e, *found = NULL;

struct list_head *list;

int h;

if (entry->rule.inode_f) {  //如果规则有对应的inode号，就通过基于节点的规则哈希链表查找规则

h = audit_hash_ino(entry->rule.inode_f->val);  //计算哈希值

*p = list = &audit_inode_hash[h];  //得到规则链表

} else if (entry->rule.watch) {    //    没有inode号，但有审计监视信息（说明规也针对着一个文件）

    //遍历基于节点的规则哈希链表，获得所有的规则链表，从中查找规则

for (h = 0; h < AUDIT_INODE_BUCKETS; h++) {

list = &audit_inode_hash[h];

list_for_each_entry(e, list, list)

if (!audit_compare_rule(&entry->rule, &e->rule)) {  //比较两条规则是否完全一致

found = e;

goto out;

}

}

goto out;

} else {        //在规则过滤链表中查找规则。查找性能不如上述算法。

*p = list = &audit_filter_list[entry->rule.listnr];

}

list_for_each_entry(e, list, list)

if (!audit_compare_rule(&entry->rule, &e->rule)) {  // //比较两条规则是否完全一致

found = e;

goto out;

}

out:

return found;

}
