1 generic_setxattr()


int generic_setxattr(struct dentry *dentry, const char *name, const void *value, size_t size,

     int flags)

{

        const struct xattr_handler *handler;

        if (size == 0)

                value = "";  

/* 查找指定命名空间中扩展属性的处理函数 */

        handler = xattr_resolve_name(dentry->d_sb->s_xattr, &name);

        if (!handler)

                return -EOPNOTSUPP;

         return handler->set(dentry, name, value, size, flags, handler->flags);

 }



2 xattr_resolve_name()


static const struct xattr_handler *xattr_resolve_name(const struct xattr_handler **handlers,    const char **name)

{

        const struct xattr_handler *handler;

        if (!*name)

                return NULL;

/*

 * 根据指定扩展属性所属的命名空间的名字，在超级块指定的处理程序数组中 * 依次查找各命名空间的处理程序

 * /

        for_each_xattr_handler(handlers, handler) {

                const char *n = strcmp_prefix(*name, handler->prefix);

                if (n) {

                        *name = n;

                        break;

                }

        }

        return handler;

}



3 vfs_setxattr()


int vfs_setxattr(struct dentry *dentry, const char *name, const void *value,size_t size, int flags)

{

        struct inode *inode = dentry->d_inode;

        int error;

/* 检查用户对指定的文件是否具有写权限 */

        error = xattr_permission(inode, name, MAY_WRITE);

        if (error)

                return error;

        mutex_lock(&inode->i_mutex);

/* 确定当前进程是否具有修改指定文件扩展属性的权限 */

        error = security_inode_setxattr(dentry, name, value, size, flags);

        if (error)

                goto out;

/*

 * 在不进行权限检查的情况下执行具体文件系统的setxattr()函数来对扩展  * 属性进行设置

 */

        error = __vfs_setxattr_noperm(dentry, name, value, size, flags);

out:     mutex_unlock(&inode->i_mutex);

        return error;

}



4 xattr_permission()



static int xattr_permission(struct inode *inode, const char *name, int mask)

{

        /* 如果要对不可修改为文件或只能追加的文件进行写操作，则失败 */

        if (mask & MAY_WRITE) {

                if (IS_IMMUTABLE(inode) || IS_APPEND(inode))

                        return -EPERM;

        }

        /*

         * 如果扩展属性位于security或system命名空间，VFS不对其进行限制，对 * 此类扩展属性的判断留给底层文件系统或安全模块，这里直接返回0          */

         if(!strncmp(name,XATTR_SECURITY_PREFIX,  XATTR_SECURITY_PREFIX_LEN)|!strncmp(name, XATTR_SYSTEM_PREFIX, XATTR_SYSTEM_PREFIX_LEN))

                 return 0;

          /* 如果扩展属性位于trust命名空间中，则只有特权用户才能进行访问 */

          if (!strncmp(name, XATTR_TRUSTED_PREFIX,

XATTR_TRUSTED_PREFIX_LEN)) {

                if (!capable(CAP_SYS_ADMIN))

                        return (mask & MAY_WRITE) ? -EPERM : -ENODATA;

                return 0;

          }

          /*

           * 对于user命名空间，只有普通文件和目录才具有扩展属性，对于“粘着   * 位”置位的目录来说，只有所有者和特权用户才能写入属性 

           */

         if (!strncmp(name, XATTR_USER_PREFIX, XATTR_USER_PREFIX_LEN)) {

                if (!S_ISREG(inode->i_mode) && !S_ISDIR(inode->i_mode))

                        return (mask & MAY_WRITE) ? -EPERM : -ENODATA;

                if (S_ISDIR(inode->i_mode) && (inode->i_mode & S_ISVTX) &&

                  (mask & MAY_WRITE) && !inode_owner_or_capable(inode))

                        return -EPERM;

        }

/*对于其他命名空间中的属性的判断则交由inode_permission()函数来进行*/

        return inode_permission(inode, mask);
