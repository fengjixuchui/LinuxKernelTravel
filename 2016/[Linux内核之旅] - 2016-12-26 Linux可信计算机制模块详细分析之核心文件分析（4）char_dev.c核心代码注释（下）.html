/*释放从form开始的count个设备号*/

void unregister_chrdev_region(dev_t from, unsigned count)

{

    dev_t to = from + count;

    dev_t n, next;

 /*从from 开始，循环释放count个设备号及其对应char_chrdev_struct对象所占的*内存空间*/

    for (n = from; n < to; n = next) {

        next = MKDEV(MAJOR(n)+1, 0);

        if (next > to)

            next = to;

        kfree(__unregister_chrdev_region(MAJOR(n), MINOR(n), next - n));

    }

}


/* 释放设备号并销毁cdev对象*/

void __unregister_chrdev(unsigned int major, unsigned int baseminor,

 unsigned int count, const char *name)

{

    struct char_device_struct *cd;

  /* 释放设备号*/

    cd = __unregister_chrdev_region(major, baseminor, count);

  /*销毁cdev对象*/

    if (cd && cd->cdev)

        cdev_del(cd->cdev);

  /*释放char_device_struct对象所占的内存空间*/

    kfree(cd);

static DEFINE_SPINLOCK(cdev_lock);

 

 static struct kobject *cdev_get(struct cdev *p)

 {

     struct module *owner = p->owner;

     struct kobject *kobj;

     if (owner && !try_module_get(owner))

         return NULL;

     kobj = kobject_get(&p->kobj);

     if (!kobj)

          module_put(owner);

      return kobj;

  }


 void cdev_put(struct cdev *p)

{

    if (p) {

        struct module *owner = p->owner;

        kobject_put(&p->kobj);

        module_put(owner);

    }

}


/* 打开字符设备文件*/

static int chrdev_open(struct inode *inode, struct file *filp)

{

    struct cdev *p;

    struct cdev *new = NULL;

    int ret = 0;

    spin_lock(&cdev_lock);

    p = inode->i_cdev;

    if (!p) {

        int idx;

        spin_unlock(&cdev_lock);

    /*搜索包括该设备号在内的范围。如果该范围不存在，则返回一个错误码，否则计算与该范围相对应的cdev描述符地址*/

        kobj = kobj_lookup(cdev_map, inode->i_rdev, &idx);

        if (!kobj)

            return -ENXIO;

  new = container_of(kobj, struct cdev, kobj);

       spin_lock(&cdev_lock);

   /*将inode对象i_cdev字段设为cdev描述符地址*/

        p = inode->i_cdev;

        if (!p) {

            inode->i_cdev = p = new;

            list_add(&inode->i_devices, &p->list);

            new = NULL;

        } else if (!cdev_get(p))

            ret = -ENXIO;

    } else if (!cdev_get(p))

         ret = -ENXIO;

    spin_unlock(&cdev_lock);

    cdev_put(new);

    if (ret)

         return ret;

     ret = -ENXIO;

  /*将文件结构file中的f_op字段初始化为cdev中的ops字段的值*/

    filp->f_op = fops_get(p->ops);

    if (!filp->f_op)

          goto out_cdev_put;

  /*fiip对象的f_op中如果定义了open方法，将open打开*/

    if (filp->f_op->open) {

          ret = filp->f_op->open(inode, filp);

          if (ret)

             goto out_cdev_put;

    }

    return 0;

    out_cdev_put:

    cdev_put(p);

    return ret;

}


/*删除cdev字符设备实例*/

void cd_forget(struct inode *inode)

{

    spin_lock(&cdev_lock);

    list_del_init(&inode->i_devices);

    inode->i_cdev = NULL;

    spin_unlock(&cdev_lock);

}


/*向系统中添加一个字符设备*/

int cdev_add(struct cdev *p, dev_t dev, unsigned count)

{

    p->dev = dev;

    p->count = count;

 /*创建一个probe对象并返回*/

    return kobj_map(cdev_map, dev, count, NULL, exact_match, exact_lock, p);

}


/*销毁probe对象*/

static void cdev_unmap(dev_t dev, unsigned count)

{

    kobj_unmap(cdev_map, dev, count);

}


/* 从系统中删除一个cdev对象*/

void cdev_del(struct cdev *p)

{

 /*销毁cdev对象对应的probe对象*/

    cdev_unmap(p->dev, p->count);

 /*将kobject对象引用计数器减1*/

    kobject_put(&p->kobj);

}


/*为cdev对象分配内存空间*/

struct cdev *cdev_alloc(void)

{

  /*分配sizeof(struct cdev)大小的内存空间*/

    struct cdev *p = kzalloc(sizeof(struct cdev), GFP_KERNEL);

    if (p) {

        /*将cdev对象插入链表*/

        INIT_LIST_HEAD(&p->list);

  /*初始化kobject 对象*/

        kobject_init(&p->kobj, &ktype_cdev_dynamic);

    }

  /*返回cdev 对象指针*/

    return p;

}


/*初始化cdev对象，将cdev对象与其对应的file_operations对象绑定*/

void cdev_init(struct cdev *cdev, const struct file_operations *fops)

{

  /*已申请内存空间初始化，此处将cdev空间中的从开始起始的sizeof *cdev 个字符初始化为0*/

    memset(cdev, 0, sizeof *cdev);

 /*链表初始化*/

    INIT_LIST_HEAD(&cdev->list);

  /初始化cdev对象对应的kobject对象*/

    kobject_init(&cdev->kobj, &ktype_cdev_default);

  /*将cdev对象与其对应的file_operations对象绑定*/

    cdev->ops = fops;

}
