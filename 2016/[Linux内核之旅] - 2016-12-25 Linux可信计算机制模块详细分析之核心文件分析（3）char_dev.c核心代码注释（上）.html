/*定义了一个kobj_map 结构体指针，kobj_map结构体用于记录设备驱动程序，它内嵌了55个struct probe指针数组*/

static struct kobj_map *cdev_map;


/*定义一个互斥量*/

static DEFINE_MUTEX(chrdevs_lock);


/*散列表chrdevs记录已经分配的设备号，散列表中每个成员是一个char_device_struct

结构体。chrdevs中包含了255（CHRDEV_MAJOR_HASH_SIZE的值）个char_device_struct 对象，每个散列表对应一个主设备号。如果分配了一个设备号,就会创建一个struct char_device_struct 对象,并将其添加到 chrdevs 中。这样,通过chrdevs数组就可以知道分配了哪些设备号*/

static struct char_device_struct {

struct char_device_struct *next;

unsigned int major;

unsigned int baseminor;

int minorct;

char name[64];

struct cdev *cdev;/* will die */

} *chrdevs[CHRDEV_MAJOR_HASH_SIZE];


/*返回chrdevs散列表的散列键值 */

static inline int major_to_index(unsigned major)

{

return major % CHRDEV_MAJOR_HASH_SIZE;

}


/* 注册设备号，函数功能是新分配一个char_device_struct 结构并对其初始化 */

static struct char_device_struct *

__register_chrdev_region(unsigned int major, unsigned int baseminor,

   int minorct, const char *name)

{

struct char_device_struct *cd, **cp;

int ret = 0;

int i;

/*分配char_device_struct 结构*/

cd = kzalloc(sizeof(struct char_device_struct), GFP_KERNEL);

if (cd == NULL)

return ERR_PTR(-ENOMEM);

   /*锁定互斥量 */

mutex_lock(&chrdevs_lock);

   /*如果主设备号major为0，请求动态分配一个主设备号，函数从散列表的末尾开

始向后寻找一个与尚未使用的主设备号对应的空链表，找到后将链表序号值i赋给主设备号major*/

if (major == 0) {

for (i = ARRAY_SIZE(chrdevs)-1; i > 0; i--) {

if (chrdevs[i] == NULL)

break;

}

if (i == 0) {

ret = -EBUSY;

goto out;

}

major = i;

ret = major;

}

   /* 初始化char_device_struct结构中的各个字段*/

cd->major = major;

cd->baseminor = baseminor;

cd->minorct = minorct;

strlcpy(cd->name, name, sizeof(cd->name));

    /*计算与主设备号对应的散列表键值*/

i = major_to_index(major);

    /*遍历散列表*/

if ((*cp)->major > major ||

    ((*cp)->major == major &&

     (((*cp)->baseminor >= baseminor) ||

      ((*cp)->baseminor + (*cp)->minorct > baseminor))))

break;

/*如果散列表中有与请求的设备号范围重叠的范围，返回错误 */

if (*cp && (*cp)->major == major) {

int old_min = (*cp)->baseminor;

int old_max = (*cp)->baseminor + (*cp)->minorct - 1;

int new_min = baseminor;

int new_max = baseminor + minorct - 1;

/* New driver overlaps from the left.  */

if (new_max >= old_min && new_max <= old_max) {

ret = -EBUSY;

goto out;

}

/* New driver overlaps from the right.  */

if (new_min <= old_max && new_min >= old_min) {

ret = -EBUSY;

goto out;

}

}

   /*将申请的char_device_stuct 结构插入到散列表中*/

cd->next = *cp;

*cp = cd;

mutex_unlock(&chrdevs_lock);

return cd;

/*错误处理，解除互斥量，释放char_chrdev_struct结构占有的内存空间*/

out:

mutex_unlock(&chrdevs_lock);

kfree(cd);

return ERR_PTR(ret);

}


/*释放设备号*/

static struct char_device_struct *

__unregister_chrdev_region(unsigned major, unsigned baseminor, int minorct)

{

struct char_device_struct *cd = NULL, **cp;

int i = major_to_index(major);

    /*锁定互斥量*/

mutex_lock(&chrdevs_lock);

    /*遍历散列表，找到要释放的设备号对应的char_chrdev_struct结构*/

for (cp = &chrdevs[i]; *cp; cp = &(*cp)->next)

if ((*cp)->major == major &&

    (*cp)->baseminor == baseminor &&

    (*cp)->minorct == minorct)

break;

/*释放找到的char_chrdev_struct结构*/

if (*cp) {

cd = *cp;

*cp = cd->next;

}

 /*解锁互斥量*/

mutex_unlock(&chrdevs_lock);

return cd;

}


/*向系统注册设备号，成功返回0，失败则返回错误代码*/

int register_chrdev_region(dev_t from, unsigned count, const char *name)

{

struct char_device_struct *cd;

dev_t to = from + count;

dev_t n, next;

for (n = from; n < to; n = next) {

next = MKDEV(MAJOR(n)+1, 0);

if (next > to)

next = to;

        /*调用__register_chrdev_region()函数注册设备号*/

cd = __register_chrdev_region(MAJOR(n), MINOR(n),

       next - n, name);

if (IS_ERR(cd))

goto fail;

}

return 0;

/*如果申请设备号失败，释放已经申请的设备号并释放设备号对应的char_chrdev_struct结构所占的内存空间*/

fail:

to = n;

for (n = from; n < to; n = next) {

next = MKDEV(MAJOR(n)+1, 0);

kfree(__unregister_chrdev_region(MAJOR(n), MINOR(n), next - n));

}

return PTR_ERR(cd);

}


/* 向系统注册设备号，主设备号由系统动态地自动分配 */

int alloc_chrdev_region(dev_t *dev, unsigned baseminor, unsigned count,

const char *name)

{

struct char_device_struct *cd;

    /*调用__register_chrdev_region()函数注册设备号*/

cd = __register_chrdev_region(0, baseminor, count, name);

if (IS_ERR(cd))

return PTR_ERR(cd);

*dev = MKDEV(cd->major, cd->baseminor);

return 0;

}


/**

*向系统申请一个固定的设备号范围，该范围包含唯一一个主设备号以及0~255的次设

*备号*/

int __register_chrdev(unsigned int major, unsigned int baseminor,

      unsigned int count, const char *name,

      const struct file_operations *fops)

{

struct char_device_struct *cd;

struct cdev *cdev;

int err = -ENOMEM;

    /*调用__register_chrdev_region()函数申请设备号*/

cd = __register_chrdev_region(major, baseminor, count, name);

if (IS_ERR(cd))

return PTR_ERR(cd);

    /*为cdev 对象分配内存空间*/

cdev = cdev_alloc();

if (!cdev)

goto out2;

    /*初始化cdev 结构体中的owner、ops字段*/

cdev->owner = fops->owner;

cdev->ops = fops;

    /*将设备驱动的名称拷贝到内嵌的kobject结构的name字段中*/

kobject_set_name(&cdev->kobj, "%s", name);

     /*向系统中添加cdev 对象*/

err = cdev_add(cdev, MKDEV(cd->major, baseminor), count);

if (err)

goto out;

    /将char_device_struct结构体中的cdev 字段指向已经存在的cdev对象*/

cd->cdev = cdev;

return major ? 0 : cd->major;

out:

    /*将kobject对象引用计数器值减1*/

kobject_put(&cdev->kobj);

   /*释放已经申请的设备号并释放设备号对应char_chrdev_struct 对象所占的内存空间*/

kfree(__unregister_chrdev_region(cd->major, baseminor, count));

return err;

}
