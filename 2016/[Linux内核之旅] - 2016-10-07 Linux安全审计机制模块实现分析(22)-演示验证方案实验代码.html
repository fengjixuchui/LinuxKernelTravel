1.内核模块代码audit_mod.c

#include <asm/uaccess.h>

#include <linux/module.h>

#include <linux/types.h>

#include <linux/init.h>

#include <asm/current.h>

#include <linux/sched.h>

#include <linux/time.h>

#include <linux/kernel.h>

 

#define COMM_SIZE   16

 

struct syscall_buf {

    u32 serial;

    u32 ts_sec;

    u32 ts_micro;

    u32 syscall;

    u32 status;

    pid_t pid;

    uid_t uid;

    u8 common[COMM_SIZE];

};

 

DECLARE_WAIT_QUEUE_HEAD(buffer_wait);

 

struct timeval xtime;

 

#define AUDIT_BUF_SIZE  100

static struct syscall_buf audit_buf[AUDIT_BUF_SIZE];

static int current_pos = 0;

static u32 serial = 0;

 

void syscall_audit(int syscall, int return_status)

{

    struct syscall_buf *ppb_tmp;

    do_gettimeofday(&xtime);

 

    if (current_pos < AUDIT_BUF_SIZE) {

        ppb_tmp = &audit_buf[current_pos];

        ppb_tmp->serial = serial++;

        ppb_tmp->ts_sec = xtime.tv_sec;

        ppb_tmp->ts_micro = xtime.tv_usec;

        ppb_tmp->syscall = syscall;

        ppb_tmp->status = return_status;

        ppb_tmp->pid = current->pid;

    }

 

    memcpy(ppb_tmp->common, current->comm, COMM_SIZE);

 

    if (++current_pos == AUDIT_BUF_SIZE * 8 / 10) {

        printk("IN module_audit: yes it near full\n");

        wake_up_interruptible(&buffer_wait);

    }

}

 

int sys_audit(u8 type, u8 *us_buf, u16 us_buf_size, u8 reset)

{

    int ret = 0;

    if (!type) {

        if (__clear_user(us_buf, us_buf_size)) {

            printk("Error: clear_user\n");

            return 0;

        }

        printk("In module_systemcall: starting...\n");

        ret = wait_event_interruptible(buffer_wait, current_pos >= \

                AUDIT_BUF_SIZE * 8 / 10);

        printk("In module_systemcall: over,current_pos:%d\n", current_pos);

        if (__copy_to_user(us_buf, audit_buf, (current_pos) * sizeof(struct \

                        syscall_buf))) {

            printk("Error: copy\n");

            return 0;

        }

        ret = current_pos;

        current_pos = 0;

    }

    printk("sys_audit done\n");

    return ret;

}

 

extern void (*my_audit)(int, int);

extern int (*my_sysaudit)(unsigned char, unsigned char *, unsigned short, \

        unsigned char);

static int __init audit_init(void)

{

    my_sysaudit = sys_audit;

    my_audit = syscall_audit;

    printk("Starting System Call Auditing\n");

    return 0;

}

 

static void __exit audit_exit(void)

{

    my_audit = NULL;

    my_sysaudit = NULL;

    printk("Exiting System Call Auditing\n");

    return;

}

 

MODULE_LICENSE("GPL");

 

module_init(audit_init);

module_exit(audit_exit);

3.用户空间测试代码auditd.c

#include <stdlib.h>

#include <stdio.h>

#include <errno.h>

#include <signal.h>

#include <stdlib.h>

#include <stdint.h>

#include <sys/resource.h>

#include <sys/syscall.h>

 

#define COMM_SIZE 16

#define __NR_myaudit 222

struct syscall_buf {

    uint32_t serial;

    uint32_t ts_sec;

    uint32_t ts_micro;

    uint32_t syscall;

    uint32_t status;

    pid_t pid;

    uid_t uid;

    unsigned char comm[COMM_SIZE];

};

 

#define AUDIT_BUF_SIZE (100 * sizeof(struct syscall_buf))

 

int main(int argc, char **argv)

{

    unsigned char col_buf[AUDIT_BUF_SIZE];

    unsigned char reset = 1;

 

    int num = 0;

    int i = 0;

    struct syscall_buf *p = NULL;

 

    while (1) {

        num = syscall(__NR_myaudit, 0, col_buf, AUDIT_BUF_SIZE, reset);

        printf("num: %d\n", num);

        if (num > 100) {

            printf("error: the value of num > 100\n");

            return -1;  

        }    

        unsigned char j = 0;

        struct syscall_buf *p;

        p = (struct syscall_buf *)col_buf;

        for (i = 0; i < num; i++) {

            printf("num[%d], serial: %d\n", i, p[i].serial);

            printf("syscall: %d\n", p[i].syscall);

            printf("ts_sec: %d\n", p[i].ts_sec);

            printf("status: %d\n", p[i].status);

            printf("pid %d\n", p[i].pid);

            printf("uid %d\n", p[i].uid);

            printf("comm %s\n", p[i].comm);

        }

    }

    return 0;

}
